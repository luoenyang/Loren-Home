# 解决树状表格复选框不联动

根据业务需求进行修改

// 手动选中某一行时
handleSelection(table, selection, row) {
    // checkAll 保存已选择的行数据

    let me = this;
    // 获取当前是否选中
    let cIndex = selection.findIndex((item => {
        return item.id == row.id;
    }))
    let isChecked = cIndex >= 0 ? true : false;

    // 获取当前行是否在已保存的数据中
    let dIndex = this.checkAll.findIndex((item => {
        return item.id == row.id;
    }));

    // 选中时，没有则添加
    if (isChecked && dIndex == -1) {
        this.checkAll.push(row);
    }
    // 取消选中时，有则删除
    else if (!isChecked && dIndex != -1) {
        this.checkAll.splice(dIndex, 1);
    }
 

    // 设置子节点选中状态和保存删除数据
    setChild(row.children);
    function setChild(children) {
        let childrens = children || [];
        childrens.forEach(item => {
            me.$refs[table].toggleRowSelection(item, isChecked);
            let cIndex = me.checkAll.findIndex((c => {
                return c.id == item.id;
            }));
            if (isChecked && cIndex == -1) {
                me.checkAll.push(item);
            }
            else if (!isChecked && cIndex != -1) {
                me.checkAll.splice(cIndex, 1);
            }


            if (item.children && item.children.length > 0) {
                setChild(item.children);
            }
        })
    }
},


// 手动点击全选时
selectAll(table, selection) {
    // checkAll 保存已选择的行数据

    let me = this;

    // 获取是否选中
    let isCheck = false;
    selection.forEach(item => {
        if (item.children && item.children.length >= 0) {
            isCheck = true;
        }
    })

    // 设置全选中和保存数据
    if (isCheck) {
        this.checkAll = [];
        getItems(this.tableData);
        function getItems(items) {
            items.forEach(item => {
                me.checkAll.push(item);
                me.$refs[table].toggleRowSelection(item, isCheck);
                if (item.children && item.children.length > 0) {
                    getItems(item.children);
                }
            })
        }
    }
    // 清除
    else {
        this.checkAll = [];
        this.$refs[table].clearSelection();
    }
},
