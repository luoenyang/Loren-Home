# 亮点代码

# Set在 JavaScript 中，Set 是一个内置对象，它允许你存储任何类型的唯一值

// 定义一个函数，用于生成指定范围内的不重复随机数
function getRandomNumbers(min, max, n) {
    let numbers = new Set();
    while (numbers.size < n) {
        // Math.floor 向下取整
        // Math.ceil 向上取整
        // Math.random>=0.0小于1
        // Math.floor(Math.random() * (max - min + 1)) 目的是让它的值最大是等于max - min，最小为0
        const randomNumber = Math.floor(Math.random() * (max - min + 1)) + min;
        // 添加
        numbers.add(randomNumber);
        // 删除
        // numbers.add(value)
        // 查询
        // numbers.has(value)
    }

    // set转成数组
    return Array.from(numbers);
}
 
// 示例：生成10个1到100之间的不重复随机数
const randomNumbers = getRandomNumbers(1, 100, 10);
console.log(randomNumbers);

# 事件委托

减少了事件监听器的数量，提高了性能，动态添加的元素，无需为每个新元素单独添加事件监听器

// DOMContentLoaded事件 完全加载后执行
document.addEventListener('DOMContentLoaded', (event) => {  
    // 获取父容器元素  
    const parentElement = document.getElementById('parent');  
  
    // 父容器中添加子元素（为了演示效果）  
    for (let i = 0; i < 10; i++) {  
        const childElement = document.createElement('div');  
        childElement.className = 'child';  
        childElement.textContent = `Child ${i + 1}`;  
        parentElement.appendChild(childElement);  
    }  
  
    // 为父容器添加点击事件监听器，实现事件委托  
    parentElement.addEventListener('click', (event) => {  
        // 检查点击的目标元素是否是子元素  
        if (event.target && event.target.classList.contains('child')) {  
            alert(`You clicked on ${event.target.textContent}`);  
        }  
    });  
  
    // 动态添加新子元素（验证事件委托）  
    const newChildElement = document.createElement('div');  
    newChildElement.className = 'child';  
    newChildElement.textContent = 'New Child';  
    parentElement.appendChild(newChildElement);  
});
