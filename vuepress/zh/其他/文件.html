var url = window.URL.createObjectURL(file); // file 、blob


// 将url加载转为文件： 本地文件是不可以加载的，但通过createObjectURL转化的就可以，因为浏览器有内存管理
fetch(url)
.then(response => response.blob())       // 将 Response 对象转换为 Blob 对象  
.then(blob => {
    var file = new File([blob], 'test'); // 将 Blob 对象转换为 File 对象
    // 使用 file 进行后续操作  
    console.log('file', file)
});

# 将文件转为base64

将文件转为base64
function readFileAsBase64(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = function () {
            resolve(reader.result);
        };
        reader.onerror = function () {
            reject(reader.error);
        };
        reader.readAsDataURL(file);
    });
}

// 将文件转base64
var base64Data = await readFileAsBase64(file);

# 将base64转Blob

将base64转Blob
// 将base64转Blob
function base64ToBlob(base64Data, contentType) {
    contentType = contentType || '';
    var sliceSize = 1024;
    var byteCharacters = atob(base64Data.split(',')[1]);
    var bytesLength = byteCharacters.length;
    var slicesCount = Math.ceil(bytesLength / sliceSize);
    var byteArrays = new Array(slicesCount);

    for (var sliceIndex = 0; sliceIndex < slicesCount; ++sliceIndex) {
        var begin = sliceIndex * sliceSize;
        var end = Math.min(begin + sliceSize, bytesLength);

        var bytes = new Array(end - begin);
        for (var offset = begin, i = 0; offset < end; ++i, ++offset) {
            bytes[i] = byteCharacters[offset].charCodeAt(0);
        }
        byteArrays[sliceIndex] = new Uint8Array(bytes);
    }
    return new Blob(byteArrays, { type: contentType });
}

var blob = base64ToBlob(base64Data, 'image/jpeg'); // 'image/png'

    var base64String = base64Data.split('base64,')[1];
    var bytes = window.atob(base64String);

# canvas转base64

canvas转base64
var url = window.URL.createObjectURL(file); // file 、blob

// 在画布上画出图片，然后把画布图片转base64
// 创建新的Image对象  
const image = new Image();
image.src = url;
// 等待图片加载完成  
image.onload = function () {
    // 创建一个canvas元素，并将图片绘制到canvas上  
    const canvas = document.createElement('canvas');
    const context = canvas.getContext('2d');
    canvas.width = image.width;
    canvas.height = image.height;
    context.drawImage(image, 0, 0, image.width, image.height);

    // 将canvas图片转换为Base64编码的字符串
    const base64Data = canvas.toDataURL('image/jpeg');
};
