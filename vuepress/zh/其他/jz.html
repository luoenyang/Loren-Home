// 获取九章系统访问授权token，需在九章后台管理界面-访问授权凭证管理添加相应凭证后得到key和secret
async function getJZFWSQToken() {
    var nonce = '123';
    var tenantname = 'ctywzypt';
    var appkey = 'afd7cf24b4bc4cd1802ae35bff9c1be3';
    var secret = '04febfe363e14788945aa4c0bedff907';
    var time = (new Date()).getTime();

    var SHA1Message = "appkey=" + appkey + "&nonce=" + nonce + "&tenantname=" + tenantname + "&timestamp=" + time;
    var SHA1Encrypt = CryptoJS.HmacSHA1(SHA1Message, secret);
    var SHA1String = SHA1Encrypt.toString();

    var aesMessage = "appkey=" + appkey + "&nonce=" + nonce + "&sign=" + SHA1String + "&tenantname=" + tenantname + "&timestamp=" + time;
    var key = CryptoJS.enc.Utf8.parse(secret);
    var AESEncrypt = CryptoJS.AES.encrypt(aesMessage, key, {
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7,
        // keySize: 256 / 32 // 256位密钥，以字节为单位
    });
    var AESString = AESEncrypt.toString();

    var v = "tenantname=" + tenantname + "&appKey=" + appkey + "&signature=" + encodeURIComponent(AESString);
    var url = "http://ywzypt.fzgxctgroup.com/v1/oauth2/getToken?" + v
    var result = await sendRequest({
        url: url,
        method: 'get'
    });

    return result;
}

async function doReportQueryV2(config) {
    var url = 'http://ywzypt.fzgxctgroup.com/report/doReportQueryV2?access_token=' + token;
    var result = {};
    await sendRequest({
        url: url,
        method: 'post',
        isFormHeader: true,
        data: {
            reportTplId: "8c365ad6-af9d-41d4-9860-687ebdbad614",
            queryString: JSON.stringify([{ "Name": "DataEnable", "DestValue": "1" }])
        }
    }).then(d => {
        result = d;
    }).catch(err => {
        result = err;
    });

    return result;
}



/**
 * 自定义拓展接口
 */
module.exports = (app) => {
    let router = new Router();

    router.get(`/getJZFWSQToken`, async (ctx, next) => {

        var result = await getJZFWSQToken();

        ctx.response.type = 'json';
        ctx.response.body = result;
    })

    router.get(`/doReportQueryV2`, async (ctx, next) => {

        var result = await doReportQueryV2();
        var times = 0;
        if (result && result.Code == 'Invalid_Login' && times == 0) {
            var t = await getJZFWSQToken();
            token = t.access_token;
            result = await doReportQueryV2();

            times++;
        }
        ctx.response.type = 'json';
        ctx.response.body = result;
    })

    // 调用路由
    app.use(router.routes());
    app.use(router.allowedMethods());
}

# 剪切板 复制到剪切板

剪切板
// 复制元素标签的文本
function getTextContentRecursive(el) {
    let textContent = '';
    const nodes = el.childNodes;
    for (let i = 0; i < nodes.length; i++) {
        const node = nodes[i];
        if (node.nodeType === Node.TEXT_NODE) {
            // 如果是文本节点，直接添加到textContent中  
            textContent += node.textContent.replace(/\n/g, '').replace(/\s+/g, ''); // 按需去除换行、空格
        } else if (node.nodeType === Node.ELEMENT_NODE) {
            // 如果是元素节点，递归处理 
            textContent += getTextContentRecursive(node);
            if (node.className == 'line' || node.className == 'subtitle') {
                textContent += '\n'; // 按需添加换行
            }
        }
    }
    return textContent;
}

// 把文本复制到剪切板
function copyToClipboard(textToCopy) {
    // navigator clipboard 需要https等安全上下文
    if (navigator.clipboard && window.isSecureContext) {
        // navigator clipboard 向剪贴板写文本
        return navigator.clipboard.writeText(textToCopy);
    } else {
        // 创建text area
        let textArea = document.createElement("textarea");
        textArea.value = textToCopy;
        // 使text area不在viewport，同时设置不可见
        textArea.style.position = "absolute";
        textArea.style.opacity = 0;
        textArea.style.left = "-999999px";
        textArea.style.top = "-999999px";
        document.body.appendChild(textArea);
        textArea.focus();
        textArea.select();
        return new Promise((res, rej) => {
            // 执行复制命令并移除文本框
            document.execCommand('copy') ? res() : rej();
            textArea.remove();
        });
    }
}

# 解决调用浏览器打印后按ESC退出打印取消打印页面的输入框获取不了焦点

  • 在打印页面按ESC退出后原页面的输入框获取不了焦点
查看
var inputEl = document.getElementById('myinput');
var printEl = document.getElementById('print');
printEl.addEventListener('click', function () {

    //该方法参数 content：需要打印的元素
    //该方法参数 content：需要打印的元素
    function printFun(content) {
        // 屏幕可用工作区宽高 
        var screenWidth = window.screen.availWidth;
        var screenHeight = window.screen.availHeight;
        var sh = window.screen.height;

        // 计算新窗口的尺寸，为视窗尺寸的50%  
        var newWindowWidth = Math.round(screenWidth * 0.8);   // 屏幕的0.8倍
        var newWindowHeight = Math.round(screenHeight * 0.8); // 屏幕的0.8倍

        var Window = window.open("",
            "页面打印", 'height=' + newWindowHeight +
            ',width=' + newWindowWidth +
            ',top=' + Math.round((screenHeight - newWindowHeight) / 2 - (sh - screenHeight)) +
            ',left=' + Math.round((screenWidth - newWindowWidth) / 2) +
        ',toolbar=no, location=no, directories=no, status=no, menubar=no, scrollbars=no, resizable=yes, copyhistory=no');
        var style = "<style type='text/css'></style>";
        Window.document.write(content + style);
        Window.focus();
        Window.document.close();     // 关闭document的输出流, 显示选定的数据
        Window.print();              // 打印当前窗口
        return Window;
    }

    var w = printFun(document.body.innerHTML);
    w.close();

    // 使用上面的取代直接调用window.print(),否则会导致上述问题
    // window.print();

    // 打印操作完成后使输入框获取焦点
    inputEl.focus();
})
