# module模块

  • 公开模块：module.exports = {}
  • 引入模块：import "module" 或者 require "module"

# 导出

// hello.js文件中
function s() {
    var name;
    this.setName = function(thyName) {
        name = thyName;
    };
    this.sayHello = function() {
        console.log('Hello ' + name);
    };
};
module.exports = s;

// 也可
// module.exports={
//     s:s
// }
