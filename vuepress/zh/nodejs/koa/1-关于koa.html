# 1. koa 是什么？

  • koa 是基于Node.js平台的web开发框架。
  • 类似于 Django 是一个开放源代码的 Web 应用框架，由 Python 写成。可作为Python的web应用框架。
  • 每门后台语言都有对的相关web框架。

# 2. koa 特点

    1. Koa 是一个新的 web 框架，由 Express 幕后的原班人马打造。
    1. 通过利用 async 函数，Koa 帮你丢弃回调函数，并有力地增强错误处理。
    1. Koa 很小的体积，并没有捆绑任何中间件。
    1. 提供的方法有如：内容协商，缓存清理，代理支持和重定向等常见任务的方法。

# 4. await next()的使用

const Koa = require('koa')
const app = new Koa()

const mid1 = async (ctx, next) => {
    ctx.body =  '前：' + '1\n'
    await next()
    ctx.body =   ctx.body + '后：' + '1\n'
}

const mid2 = async (ctx, next) => {
    ctx.body =    ctx.body + '前：'+ '2\n'
    await next()
    ctx.body =    ctx.body + '后：'+ '2\n'
}

const mid3 = async (ctx, next) => {
    ctx.body =  ctx.body + '前：'+  '3\n'
    await next()
    ctx.body =   ctx.body + '后：'+ '3\n'
}

app.use(mid1)
app.use(mid2)
app.use(mid3)

app.listen(3000)

# 5. 简单应用-路由

const Koa = require('koa');
const koaStatic = require('koa-static');
const koaBody = require('koa-body');
const Router = require('koa-router');

const app = new Koa();
app.use(koaBody());          // 或者 app.use(koaBody.koaBody());
app.use(koaStatic('./www')); // 静态资源文件夹

// 路由服务
let router = new Router();

router.get(`/test`, async (ctx, next) => {
    let param = util.getUrlParam(ctx);
    let result = { msg: '请求成功！' }
    // await next();
    await someFunction();

    ctx.response.type = 'json';
    ctx.response.body = result;
}

// 调用路由
app.use(router.routes());
app.use(router.allowedMethods());
app.listen(8888);
console.log('启动服务: http://localhost:' + 8888);
