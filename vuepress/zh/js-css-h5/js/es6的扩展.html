代码示例

for (let i = 0; i < 10; i++) { }

// 错，不存在变量提升
console.log(x1);                   
let x1;

// 错，不可重复声明
let x1=1;

const Y=123;
Y=456;

// 错，y没有被声明，改为 someFun( y = 2,x = y) 则对
function someFun(x = y, y = 2) {
    return [x, y];

# 解构赋值

能转化成对象或者数组的都可以解构

代码示例
// 数组, x1=1,y1=2,x1=3
let [x1, y1, z1] = [1, 2, 3]; 

// 对象, foo='aaa',bar='bbb'
let { foo, bar} = { foo: "aaa", bar: "bbb" }; 
let obj={
    a:'a',
    b:'b',
    c:{
        d:'d',
        e:'e',
        f:{
            g:'g'
        }
    }
}
let {a,b,c:{d,f:{g}},c:cc}=obj;

console.log(b) // b
console.log(d) // d
console.log(g) // g
console.log(cc) // obj.c 整个对象

// 字符串, a='h' ...
let [a, b, c, d, e] = 'hello';

// 字符串, len=5
let {length:len}='hello'

// 函数, 8
function add({x, y} = { x: 0, y: 0 }) {
    return x+y;
}
add({x:3,y:5})

# 字符串的扩展

代码示例

for (let i of 'foo') {
    console.log(i)   // 循环打出f, o , o
}

// 字符串的判断
var s = 'Hello world!';			
s.startsWith('world', 6)    // true   是否以什么开头，数字表示开始搜索的位置 
s.endsWith('Hello', 5)      // true   是否以什么结尾，数字表示开始搜索的位置

<div>${variable.count}</div> 
`)

   console.log(String.raw`Hi\n${2+3}!`); // 输出 "Hi\\n5!"
   console.log(String.raw`Hi\\n`)        // 输出 "Hi\\n"

// 等等

# 数值的扩展

关键词：Number.isFinite()， isFinite()， Number.isNaN()， isNaN()， Number.isInteger()，
Math.trunc， Math.sign(判断正负1 -1 0 -0)， Math.cbrt(立方根)， 2**2

代码示例
// Number.isFinite(): 只要是非数值都返回false ,
// isFinite(): 会先转为数字
Number.isFinite(NaN);     // false
Number.isFinite(true);    // false
Number.isFinite('15');    // false       
// 区别于传统   
isFinite('15')            // true

// Number.isNaN(): 只要是非NaN都返回false ,
// isNaN(): 会先转为数字
Number.isNaN(NaN)                       // true
Number.isNaN('15')                      // false
Number.isNaN(15)                        // false
Number.isNaN(true)                      // false
Number.isNaN(9/NaN)                     // true
Number.isNaN('true'/0)                  // true
Number.isNaN('true'/'true')             // true
Number.isNaN("NaN")                     // false
// 区别于传统
isNaN("NaN")                            // true

// Number.isInteger(): 判断是否是整型, 
Number.isInteger(3.0)                   // true
Number.isInteger(3)                     // true
            
// Number.EPSILON: 极小的常量
0.1 + 0.2                               // 0.30000000000000004
0.1 + 0.2 - 0.3                         // 5.551115123125783e-17
if(5.551115123125783e-17<Number.EPSILON)
{
    console.log("这个数值可以用");       // 相当于设置一个误差，如果小于这个误差就是对的
}
			
// Number.isSafeInteger(): 安全整数  +-2的53次方
Number.isSafeInteger(Infinity)          // false
Number.isSafeInteger(-Infinity)         // false

// Math对象的扩展                             
Math.trunc(5.321)                       //5 ，去除小数部分，和 parseInt差不多
Math.trunc(NaN)                         //NaN
    
Math.sign(-5)                           // -1 ，   判断是正负，0。                    
Math.sign(5)                            // +1
Math.sign(0)                            // +0
Math.sign(-0)                           // -0

Math.cbrt(27)                           // 立方根， 3
			
// 指数运算符 ** : 前值表示底，后值表示幂
2**2                                    // 4      
2**3                                    // 8

# 函数的扩展

关键词：函数的默认值，函数的默认值与解构赋值，箭头函数
箭头函数本身没有被this，this绑定: .bind()

代码示例
// 函数的默认
function log(x, y = 'World') {
    console.log(x, y);
}
log('Hello') ;                                // Hello World
log('Hello','') ;                             // Hello 
log('Hello','aaa') ;                          // Hello aaa
            
// 解构赋值与默认值结合使用
function foo({x, y = 5}) {
    console.log(x, y);
}
foo({})                                       // undefined, 5
foo({x: 1})                                   // 1, 5
foo({x: 1, y: 2})                             // 1, 2
foo()                                         // 报错
            
function f0(x, y, z) {}
var args = [0, 1, 2];
f0(...args);                                  // 对象展开符'...'不能少
        

// 箭头函数
var f = (v) => {return v};                          
//等同
var f = (v) => v;                          
var f = function(v) {
   return v;
};
			 
// 如果要返回一个对象,简写
var sum = (num1, num2) =>({num:num1+num2});
			
// 箭头函数使用注意：this ，箭头函数本身没有this
setTimeout(() => this.some);   // 此this指向它的上级对象或者全局变量
			
// this绑定: .bind()
class Myclass {
    constructor(){
        this.type = 'myclass';
        //says使用this时,指向Myclass
        this.says = this.says.bind(this); 
    }
    says(say){
        setTimeout( () => {
            console.log(this.type + ' says ' + say)
        }, 1000)
    }
}
var animal = new Animal()
animal.says('hi')  // 输出 myclass says hi

# 数组的扩展

关键词：扩展运算符，将字符串转为数组，Array.from() 对象转数组，Array.of，遍历数组 for of

代码示例

// 扩展运算符
const [one, ...last] = [1, 2, 3, 4, 5];  // 结果：one = 1,last = [2,3,4,5]
            
// 将字符串转为数组
console.log([...'123456789']);
            
// Array.from(): 将类似数组的对象转为真正的数组
let arrayLike = {
    '0': 'a',
    '1': 'b',
    '2': 'c',
    length: 3
};
// ES5的写法
var arr1 = [].slice.call(arrayLike);    // 结果： ['a', 'b', 'c']
// ES6的写法
let arr2 = Array.from(arrayLike);       // 结果： ['a', 'b', 'c']

// Array.of: 方法用于将一组值，转换为数组。
var arr3 = Array.of(3, 11, 8) ;         // 结果:  ['3', '11', '8']     
            

           
// 遍历数组，entries()键值对，keys()键 和values()值  , forEach(value,index,input)
for (let value of ['a', 'b']) {
    console.log(value); // 输出 a b
}
for (let value of ['a', 'b'].values()) {
console.log(index); // 输出 a b
}
for (let index of ['a', 'b'].keys()) {
    console.log(index); // 输出 0 1
}
for (let [index, elem] of ['a', 'b'].entries()) {
    console.log(index, elem); // 输出 0 a,    1,b
}

# 对象的扩展

关键词：属性的简写，属性表达式: [变量]，简写和表达式不可同时使用，Object.is(object1,object2)
对象的合并，遍历对象 for in ，对象展开符

代码示例

// 属性的简洁表示法
// 例1
var foo1 = 'bar';
var baz1 = {foo1};
// 例2
function f(x, y) {

  return "Hello!";

# symbol数据类型

关键词：Symbol 值不能与其他类型的值进行运算, 可以转化为字符串。
独一无二的值，不能四则运算，不能和其他类型的值换算

代码示例

// 没有参数的情况
var s1 = Symbol();
var s2 = Symbol();
s1;                               // Symbol()
s2;                               // Symbol()
s1 === s2;                        // false
			
// 有参数的情况
var s3 = Symbol('foo');
var s4 = Symbol('foo');
s1;                               // Symbol('foo')
s2;                               // Symbol('foo')
s3 === s4                         // false
			
// symbol参数可以是一个对象
var s5={
    toString(){return "s5"}       // toString() , 如果不这样写，返回的是object
};
var s6=Symbol(s5);
console.log(s6)
			
// 转化为字符串
String(s4)                        // 'Symbol(foo)'
s4.toString()                     // 'Symbol(foo)'

// 不能和其他类型的值换算
'some'+s4                         // 报错


// 作为对象的属性名 ，因为他是独一无二
var mySymbol = Symbol(); 
// 第一种写法
var a = {}; 
a[mySymbol] = 'Hello!';           // a[mySymbol] = 'Hello'
// 第二种写法
var a2 = {
    [mySymbol]: 'Hello!'          // a2[mySymbol] = 'Hello'
};

// Symbol 值作为对象属性名时，不能用点运算符。
a2.mySymbol                       // 出错，此时的mySymbol并不是Symbol值，它是一个字符串
 
 
// Symbol.for():重新使用同一个Symbol值,搜索有没有以该参数作为名称的Symbol值。
// 如果有，就返回这个Symbol值，否则就新建并返回一个以该字符串为名称的Symbol值。
var s0 = Symbol('foo');
var s1 = Symbol.for('foo');
var s2 = Symbol.for('foo');			
s1 === s2  // true
s0 === s1  // false
           
// Symbol.keyFor方法返回一个已登记的 Symbol 类型值的key。Symbol.for()类型的。
Symbol.keyFor(s0);                                             // undefined
Symbol.keyFor(s1);                                             // foo
Symbol.keyFor(s2);                                             // foo

# Set和Map数据结构

关键词：

  • Set：类似数组 值的集合，但不能用下标取值，成员唯一不重复，.add .delete .has .clear .size，可转数组
  • Map：类似对象 键值对，不能用点取值，.set .get .has .delete .clear .size
  • Map可以通过get方法获取值，而set不能因为它只有值，set只能用has来判断，返回一个布尔值
  • Set的值是唯一的可以做数组去重，Map由于没有格式限制，可以做数据存储
代码示例
// Set(): add()添加,没有重复
let set = new Set([1, 2, 3, 4, 4]);
set.add(4);
console.log([...set]);      // [1, 2, 3, 4]              
console.log(set.size);      // 4 , 用size 而不是length

let a = NaN;
let b = NaN;
set.add(a);
set.add(b);                                               // [1, 2, 3, 4, NaN]
console.log(set[0]);                                      // undefined，还不是数组
            
            
// Array.from: 将Set转化为数组
const items = new Set([1, 2, 3, 4, 5]);
const array = Array.from(items);
console.log(array[0]);                                    // 1 ，已经是数组

# Proxy和Reflect
  • Proxy 可以理解成，在目标对象之前架设一层“拦截”，外界对该对象的访问，都必须先通过这层拦截，因此提供了一种机制，可以对外界的访问进行过滤和改写。 Vue3实现数据绑定的方式。
  • 它是作用于整个对象，区别于 Object.defineProperty作用于对象的某属性
代码示例
// 对一个空对象{}架设了一层拦截，重定义了属性的读取（get）和设置（set）行为。
var obj = new Proxy({}, {
    get: function (target, key, receiver) {
    return 36;
    },
    set: function (target, key, value, receiver) {
    return some();
    }
});

# Generator函数

关键词：function* some(){}，必须使用 next 才执行，也可以自动执行，

代码示例

yield是JS为了解决异步调用的命令。表示程序执行到这里会交出执行权，等待结果返回。它需要在协程Generator 函数中运行。

// Generator 函数
function* helloWorldGenerator() {
    yield 'hello';
    yield 'world';
    yield 123+123;                 // 不会立即求值，只会在next方法将指针移到这一句时，才会求值。
    yield 'Hello' + (yield 123);
    let my=yield 'myyield';
    console.log(my);               // undefined
    return 'ending';
}
// 返回的遍历器对象，可以依次遍历
var hw = helloWorldGenerator();    // 必须 , helloWorldGenerator.next() 会出错
// 必须调用next方法,hw才会执行
console.log(hw.next())             // { value: 'hello', done: false }
console.log(hw.next())             // { value: 'world', done: false }
console.log(hw.next())             // { value: 246, done: false }
console.log(hw.next())             // { value: '123', done: false }
console.log(hw.next())             // { value: helloundefined, done: false }
console.log(hw.next())             // { value: myyield, done: false }  注意，他返回的是yield的状态，并不是my , my=undefined
console.log(hw.next())             // { value: ending, done: false }
			
//  next()参数：next方法可以带一个参数，该参数就会被当作上一个yield表达式的返回值。
function* f() {
    for(var i = 0; true; i++) {
    var reset = yield i;           // next() 没有参数时 ，reset总是等于 undefined
    if(reset) { i = -1; }          // 注意他是没有return的, 只是赋值
    }
}
var g = f();
console.log(g.next())              // { value: 0, done: false }
console.log(g.next())              // { value: 1, done: false }
console.log(g.next(true))          // { value: 0, done: false } , 
                                   // true被当做上次yield的返回值,所以reset = true,下次循环的时候i从-1开始,所以i++ = 0

 
// yield* : 用来在一个 Generator 函数里面执行另一个 Generator 函数。
           
// yield*后面跟数组 和 字符串  、对象
let read = (function* () {
    yield* ["a", "b", "c"];
    yield 'hello';
    yield* 'hello';
})();

console.log(read.next().value)              // "a"
console.log(read.next().value)              // "b"
console.log(read.next().value)              // "c"
console.log(read.next().value)              // "hello"
console.log(read.next().value)              // "h"


// Generator 函数可以自动执行。
function* gen9() {
    // ...
}
var g = gen9();
var res = g.next();

while(!res.done){
    console.log(res.value);
    res = g.next();
}

// 利用 for of 循环自动执行
let reader=read();
for(const readItem of reader){
    console.log(readItem)
}

# js 的几种模块规范？

  • 第一种是 CommonJS 方案，它通过 require 来引入模块，通过 module.exports 定义模块的输出接口。这种模块加载方案是服务器端的解决方案，它是以同步的方式来引入模块的，因为在服务端文件都存储在本地磁盘，所以读取非常快，所以以同步的方式加载没有问题。但如果是在浏览器端，由于模块的加载是使用网络请求，因此使用异步加载的方式更加合适。

  • 第二种是 AMD 方案，这种方案采用异步加载的方式来加载模块，模块的加载不影响后面语句的执行，所有依赖这个模块的语句都定义在一个回调函数里，等到加载完成后再执行回调函数。require.js 实现了 AMD 规范。

  • 第三种是 CMD 方案，这种方案和 AMD 方案都是为了解决异步模块加载的问题，sea.js 实现了 CMD 规范。它和require.js的区别在于模块定义时对依赖的处理不同和对依赖模块的执行时机的处理不同。

  • 第四种方案是 ES6 提出的方案 ES-Module，使用 import 和 export 的形式来导入导出模块。
