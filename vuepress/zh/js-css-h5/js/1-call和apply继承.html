# call和apply继承

查看
// 继承的作用：继承一个对象（函数）的方法等，减少代码量。

// call() 方法和apply() 方法的作用相同：改变this指向



// call 定义：调用一个对象的一个方法，以另一个对象替换当前对象。
// apply 定义：应用某一对象的一个方法，用另一个对象替换当前对象。



// call与aplly的异同：

// 1. 第一个参数this都一样,指当前对象
// 2. 第二个参数不一样：call的是一个个的参数列表；apply的是一个数组



// 源函数
function Person(name, age, love) {
    this.name = name;
    this.age = age;
    this.love = love;
    this.say = function say() {
        console.log(name + " love " + love);
    }
}

// call方式
function Call(name, age, love) {
    Person.call(this, name, age, love);
}

// apply方式
function Apply(name, age, love) {
    Person.apply(this, [name, age, love]);
    //Person.apply(this,arguments); //跟上句一样的效果，arguments
}



console.log("两种继承方式：")
var per = new Person("源函数", 20, "源老婆");            //输出：源函数 love 源老婆
per.say();
var call = new Call("Call 继承", 18, "Call 老婆");       //输出：Call 继承 love Call 老婆
call.say();
var apply = new Apply("Apply 继承", 16, "Apply 老婆");   //输出：Apply 继承 love Apply 老婆
apply.say();






// apply的妙用 ：
// 1. 获取数组最大值
// 2. 两个数组的连接追加并改变前面的数组 有别于concat()

var arr = [5, 7, 9, 1];
var arr2 = [3, 4, 5];
console.log("获取数组最大值: " + Math.max.apply(null, arr))
// 输出 9
console.log("两个数组连接 返回总个数7: " + Array.prototype.push.apply(arr, arr2));
// 输出 7
console.log('追加后的arr', arr);
// 输出 [5, 7, 9, 1,3, 4, 5]





// javascript max 的使用， max() 只能用数字 不能用数组
console.log("比较数值大小获取最大值: " + Math.max(1, 20, 60, 30, 10));
// 输出 60

# bind call apply 区别

都是来改变this指向和函数的调⽤，即继承

查看
# 1. call 方法

语法：函数名.call(调用者, 参数1, …) 作用：函数被借用时，会立即执行，并且函数体内的this会指向借用者或调用者

function fn(name, age) {
    this.name = name;
    this.age = age;
}
const obj = {}
// 经过call，this指向obj，obj开始有name、age属性
fn.call(obj, '李四', 100)
// 以下this均指向window,name:undefined,age:undefined
fn.call()
fn.call(null)
fn.call(undefined)

# 2. apply方法

语法：函数名.apply(调用者, [参数, …]) 作用：函数被借用时，会立即执行，并且函数体内的this会指向借用者或调用者

function fn(name, age) {
    this.name = name;
    this.age = age;
}
const obj = {}
// 经过apply，this指向obj，obj开始有name、age属性
fn.apply(obj, ['李四', 100])
// 以下this均指向window,name:undefined,age:undefined
fn.apply()
fn.apply(null)
fn.apply(undefined)

# 3. bind方法

语法：函数名.bind(调用者, 参数, …) 作用：函数被借用时，不会立即执行，而是返回一个新的函数。需要自己手动调用新的函数来改变this指向

function fn(name, age) {
    this.name = name;
    this.age = age;
}
const obj = {}
// const newFn = fn.bind(obj, '李四', 100)
// newFn()
fn.bind(obj, '李四', 100)()

# 总结：

  • 相同点： 三者都可以把一个函数应用到其他对象身上，注意不是自身对象
  • 不同点：
    • call，apply是直接执行函数调用。bind是绑定，执行需要再次调用。
    • call，bind接收逗号分隔的无限个参数列表；apply接收数组作为参数。
