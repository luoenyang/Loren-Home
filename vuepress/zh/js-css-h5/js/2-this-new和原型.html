# this、new、和原型的关系

关系

  1. this的指向在函数定义的时候是确定不了的，只有函数执行的时候才能确定this到底指向谁，实际上this的最终指向的是那个调用它的对象。

  2. new的作用是通过构造函数来创建一个实例对象。

  3. 原型链主要 为了共享属性和继承。

  • 原型链：当一个对象需要调用某个方法时，它会去最近的原型上查找该方法，如果没有找到，它会再次往下继续查找。这样逐级查找，一直找到了要找的方法。 这些查找的原型构成了该对象的原型链条。

# 原型对象

  • 原型对象存在函数中。
  • prototype为函数对象的一个属性，这个属性指向原型对象; prototype 只存在函数中;
  • _proto_这个属性指向创建它的引用类型的原型对象; 不仅在函数 proto 也存在对象类型中。
代码示例

# this的指向

    1. 普通函数的this：指向它的调用者，如果没有调用者则默认指向window。
    1. ES5中普通函数this指向window，ES6中普通函数this指向为undefined。
    1. 事件的this指向，为事件源本身。
    1. 定时器的this指向，为window。
    1. 构造函数内部的this指向new出来的这个对象。
    1. 箭头函数的this: 指向箭头函数定义时所处的对象，而不是箭头函数使用时所在的对象（箭头函数本身没有this），默认使用父级的this。
代码示例
// 示例1 在对象中，在普通函数中，在箭头函数中，在定时器中
var obj1 = {
    user: "韩梅梅",
    fn1: function () {
        console.log('fn1', this, this.user);   // 普通函数 this 指向 obj1, 因为调用这个fn1是通过obj1.fn1()执行的

        // 闭包，自执行函数
        (function () {
            console.log('自执行', this)        // 最终指向 window
        })();

        function say() {
            console.log('say', this)           // 最终指向 window
        }
        say();

        (() => {
            console.log('自执行', this)        // 最终指向 obj1
        })();

        // 定时器
        setTimeout(function () { 
           console.log('setTimeout',this)      // 最终指向 window 
        }, 1000)

        return () => {
            console.log('fn1 return', this)    // 箭头函数 this 指向 父层 或者 window，此处指向 obj1
        }
    },
    fn1_1: function () {
        console.log('fn1_1', this)             // 普通函数 this 指向 obj1, 因为调用这个fn1是通过obj1.fn1()执行的 
        return function f() {
            console.log('fn1_1 return', this)  // 指向window
        }
    },
    // 和fn1等效
    fn2() {
        console.log('fn2', this)               // 最终指向 obj1
        return () => {
            console.log('fn2 return', this)    // 箭头函数 this 指向 父层 或者 window，此处指向 obj1
        }
    },
    fn3: () => {
        console.log('fn3', this);              // 箭头函数 this 指向 父层 或者 window，此处指向 window
        return () => {
            console.log('fn3 return', this)    // 箭头函数 this 指向 父层 或者 window，此处指向 window
        }
    },
}
obj1.fn1()();
obj1.fn1_1()();
obj1.fn2()();
obj1.fn3()();


// 示例2 箭头函数绑定候无法修改
function t() {
    return () => {
        console.log(this.a);        // this 指向 t = {a:2}
    };
}
var test1 = t.call({ a: 2 });       // this 指向 test1
test1();                            // 打印：2
var test2 = test1.call({ a: 3 });   // 尝试修改箭头函数中的为3，箭头函数的this，绑定后无法修改。
test2();                            // 打印：2



// 示例3 构造函数中的this
function c1(name) {
    if (name) {
        this.name = name;           // this 指向 t = {name:name(实参)}
    }

    // 注释掉上面，直接打印 console.log(this)
    // 输出 {}                      // this 指向 new c1() 对象   
} 

function c2(name) {
    this.name = name; // 如果注释掉这一行，那么new c2().name 就是李雷
}

function c3(name) {
    this.name = name || "韩梅梅"; 
}

c1.prototype.name = "李雷";
c2.prototype.name = "李雷"; // 初始化时已给定了name(undefined),此时改不了它的参数了
c3.prototype.name = "李雷";

// 说明：prototype为函数对象的一个属性，这个属性指向原型对象; prototype 只存在函数中;
// 说明：_proto_这个属性指向创建它的引用类型的原型对象; 不仅在函数 _proto_ 也存在对象类型中

// c1.prototype == new c1().__proto__;
// c2.prototype == new c2().__proto__;
// c3.prototype == new c3().__proto__;

// 注意
// c3.prototype.constructor == c3;


// 下面的执行过程中 this 指向了new 出来的对象

console.log(new c1(), new c2(), new c3());
console.log(new c1().name, new c2().name, new c3().name);
// 打印：李雷 undefined（先在构造函数中找name，找到了undefined，就不继续找其原型了） 韩梅梅
console.log(new c1("hello").name, new c2("嗨").name, new c3().name);
// 打印：hello 嗨 韩梅梅
console.log(c1.prototype.name, c2.prototype.name, c3.prototype.name);
// 打印：李雷 李雷 李雷

function Person(name, age, love) {
    this.name = name;

    this.love = love;

function Call(name, age, love) {
    Person.call(this, name, age, love);

function Apply(name, age, love) {
    Person.apply(this, [name, age, love]);

# new原理

  1. 新建了一个Object对象
  2. 修改构造函数this指向这个Object对象，并执行构造函数
  3. 为Object对象添加了一个proto属性，是其指向构造函数的prototype属性
  4. 返回这个Object对象
