# 获取屏幕宽高

// 获取屏幕宽度
var screenWidth = window.screen.width;

// 获取屏幕高度
var screenHeight = window.screen.height;


console.log("屏幕宽度：" + screenWidth);
console.log("屏幕高度：" + screenHeight);


// 屏幕可用工作区高度：
var availHeight = window.screen.availHeight;
// 屏幕可用工作区宽度：
var availWidth = window.screen.availWidth;
console.log("可用屏幕宽度：" + availWidth);
console.log("可用屏幕高度：" + availHeight);

# 获取浏览器窗口宽高

// 注意，这些属性返回的是浏览器窗口的实际宽度和高度，不包括浏览器工具栏、地址栏、滚动条等占用的空间

console.log("浏览器窗口实际可用高度（不含工具栏、不含调试面板）：" + height);

// 通常包括浏览器工具栏、地址栏、滚动条等占用的空间
var outerWidth = window.outerWidth;
var outerHeight =  window.outerHeight;
console.log("浏览器窗口总宽度：" + outerWidth);
console.log("浏览器窗口总高度：" + outerHeight);

# 获取元素位置

var element = document.getElementById("myElement");

// 获取元素相对浏览器视口位置和大小（不含margin，含padding和border）
element.getBoundingClientRect()

// 方法返回一个 DOMRect 对象，该对象包含了元泰的位置和大小信息。DOMRect 对象具有以下属性：

// left：元素的左边缘相对于浏览器视口的左边缘的距离，单位为像素。
// top：元泰的上边缘相对于浏览器视口的上边缘的距离，单位为像素。
// right：元素的右边缘相对于浏览器视口的左边缘的距离，单位为像素。
// bottom：元素的下边缘相对于浏览器视口的上边缘的距离，单位为像素。
// width：元素宽（不含margin）
// height: 元素高（不含margin）





// 获取元素相对位置
var offsetLeft = element.offsetLeft;
var offsetTop  = element.offsetTop;
// offsetLeft = 当前元素的 marginLeft + 最近的已定位父级（offsetParent）的 paddingLeft
// offsetTop = 当前元素的 marginTop + 最近的已定位父级（offsetParent）的 paddingTop
// 如果父级都没有定位，offsetLeft,offsetTop 是当前元素 分别到 body 顶部 和 左边 的距离。

# 获取元素大小

var element = document.getElementById("myElement");


// 获取元素大小（含边框，含内边距，不含外边距）
var width = element.offsetWidth;
var height = element.offsetHeight;
console.log('width', width)
console.log('height', height)

// 获取元素大小（不包含边框，含内边距，不含外边距）
var width2 = element.clientWidth;
var height2 = element.clientHeight;
console.log('width2', width2)
console.log('height2', height2)
