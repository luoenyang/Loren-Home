# 截取两个字符之间的值

// 示例1  匹配在一个冒号之后，到下一个冒号之前的所有字符
const str = "这是一个示例::截取这个值::这也是一个示例";
const regex = /(?<=:).*?(?=:)/;
const match = str.match(regex);
console.log(match[0]); // 输出：截取这个值


// 示例2  截取e和1之间的字符
const str = "abcde哈哈123456";
const regex = /e(.*?)1/;
const result = regex.exec(str);

if (result) {
  console.log(result[1]); // 输出：哈哈12345
} else {
  console.log("未找到匹配的字符串");
}
