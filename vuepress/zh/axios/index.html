# axios是什么

易用、简洁且高效的http库， 支持node端和浏览器端，支持Promise，支持拦截器等高级配置。
前后台均可用。

类似$.ajax
post请求： 不管请求头是json还是form，参数体key 都是 data
get请求： params:{}

# 简单请求 方式：application/x-www-form-urlencoded

const axios = require('axios');
const qs = require('qs');
const dv = {a:1,c:{d:2,f:[3,4]}};
var postData,isFormHeader=true;
if (typeof dv == 'object'&&isFormHeader) {
    for (let i in dv) {
        let v = dv[i];
        // 判断参数值是否是数组
        // dv[i] = v && Array.isArray(v) ? JSON.stringify(v) : v;
        
        // 数组[]或是对象{}
        // 将参数序列化为JSON格式
        dv[i] = v && typeof v == 'object' ? JSON.stringify(v) : v;
    }
    // 序列化为URL编码格式的字符串 使用 qs.stringify 序列化对象，并设置 allowDots: true 以便支持点符号
    postData = qs.stringify(dv,{ allowDots: true });
}
else{
    postData=dv;
}


axios({
    method: "post",
    url: "https://example.com/api",
    headers: {
       'Content-Type': isFormHeader?'application/x-www-form-urlencoded': "Content-Type": "application/json",
    },
    data: postData,  
})
    .then((response) => console.log(response.data))
    .catch((error) => console.error(error));

# 封装请求

import axios from 'axios';
import { ElMessageBox } from 'element-plus'
import qs from 'qs'; // 自带 直接引入

const httpUtil = {
    /**
     * post请求
     * 把axios的请求方式改成符合我们的使用习惯
     * @param {String} url 请求路径
     * @param {Object} params 请求参数
     * @param {Function} callback 请求成功回调
     * @param {Boolean} isJson 是否是application/json方式请求
     */

    // 回调的方式处理请求后的数据（回调函数的方式，非返回promise）
    post: function (url, params,isJson, callback) {
        let headers = {};
        let cfg = {
            url: url,
            method: 'post',
        }
        if (isJson) {
            headers['Content-Type'] = 'application/json';
            // 这个好像是不对的  都要用data
            // cfg.params = params;
            cfg.data = params;
        } else {
            headers['Content-Type'] = 'application/x-www-form-urlencoded';

            // url字符串形式提交（参数值都应转化为字符串/json字符串 如{a:1,b:2} ==> a=1&b=2）

            // ***** 使用：URLSearchParams *****
            // 在将数据转换为 URLSearchParams 时，axios 会自动将对象转换为字符串，并进行编码。
            // cfg.data = new URLSearchParams({a: 1, c: [2, 3]}).toString();

            // ***** 使用：Qs *****
            // 如果参数值是数组，需先将数组转化成json字符串，否则axios最终提交的参数格式会错误，当然也可以在传入之前就转
            // 如果参数值是数组，qs无法正常转化它会将数组进行拆分 而不是一个整体了
            // 如果是对象如{a:1}则不需要，使用qs转即可
            

            // 传入的是个对象{}
            if (typeof params == 'object') {
                for (let i in params) {
                    let v = params[i];
                    // 判断参数值是否是数组
                    // params[i] = v && Array.isArray(v) ? JSON.stringify(v) : v;

                    // 数组[]或是对象{}
                    // 将参数序列化为JSON格式
                    params[i] = v && typeof v == 'object' ? JSON.stringify(v) : v;
                }
            }
            // 序列化为URL编码格式的字符串
            cfg.data = qs.stringify(dv);         
        }
        cfg.headers = headers;

        axios.request(cfg).then(function (rs) {
            let rsData = rs.data;
            if (callback && typeof callback == 'function') {
                callback(rsData);
            }
        }).catch(function (error) {

            console.log(error);
            // 错误提示
            if (error.response) {
                if (error.response.status == '401') {
                location.href = "/#/login";
                return;
                } else {
                ElMessageBox.alert("服务器或网络错误！");
                }
            } else if (error.request) {
                ElMessageBox.alert("请求错误！");
                console.log(error.request);
            } else {
                console.log('Error', error.message);
            }

            // 回调
            if (callback && typeof callback == 'function') {
                callback(error);
            }
        });
    },
    // 返回一个promise
    asyncPost: async function (url, params,isJson) {
        let headers = {};
        let cfg = {
            url: url,
            method: 'post',
        }
        if (isJson) {
            headers['Content-Type'] = 'application/json';
            // 这个好像是不对的  都要用data
            // cfg.params = params;
            cfg.data = params;
        } else {
            // url字符串形式提交（参数值都应转化为字符串/json字符串 如{a:1,b:2} ==> a=1&b=2）
            headers['Content-Type'] = 'application/x-www-form-urlencoded';
            // 如果参数值是数组，需先将数组转化成json字符串，否则axios最终提交的参数格式会错误，当然也可以在传入之前就转
            // 如果参数值是数组，qs无法正常转化它会将数组进行拆分 而不是一个整体了
            // 如果是对象如{a:1}则不需要，使用qs转即可

            // 传入的是个对象{}
            if (typeof params == 'object') {
                for (let i in params) {
                    let v = params[i];
                    // 判断参数值是否是数组
                    // params[i] = v && Array.isArray(v) ? JSON.stringify(v) : v;

                    // 数组[]或是对象{}
                    // 将参数序列化为JSON格式
                    params[i] = v && typeof v == 'object' ? JSON.stringify(v) : v;
                }
            }
            // 序列化为URL编码格式的字符串
            cfg.data = qs.stringify(params); 
        }
        cfg.headers = headers;

        let res = await axios.request(cfg);
        if (res.data.success === true) {
            console.log(res.data)
            return res.data;
        }
        return Promise.reject(new Error(res.data));
    },

    get: function (url, params, callback) {
        // 传入的是个对象{}
        if (typeof params == 'object') {
            for (let i in params) {
                let v = params[i];
                // 判断参数值是否是数组
                // params[i] = v && Array.isArray(v) ? JSON.stringify(v) : v;

                // 数组[]或者对象{}
                // 将参数序列化为JSON格式
                params[i] = v && typeof v == 'object' ? JSON.stringify(v) : v;
            }
        }
        axios.request({
        url: url,
        method: 'get',
        params: params
        }).then(function (rs) {
        let rsData = rs.data;
        if (callback && typeof callback == 'function') {
            callback(rsData);
        }
        }).catch(function (error) {
        if (error.response) {
            ElMessageBox.alert("服务器或网络错误！");
        } else if (error.request) {
            ElMessageBox.alert("请求错误！");
            console.log(error.request);
        } else {
            console.log('Error', error.message);
        }
        });
    },

    getFile: function (url, params, callback) {
        console.log('开始下载文件：'+url);
        // 传入的是个对象{}
        if (typeof params == 'object') {
            for (let i in params) {
                let v = params[i];
                // 判断参数值是否是数组
                // params[i] = v && Array.isArray(v) ? JSON.stringify(v) : v;

                // 数组[]或者对象{}
                // 将参数序列化为JSON格式
                params[i] = v && typeof v == 'object' ? JSON.stringify(v) : v;
            }
        }
        axios.request({
            url: url,
            method: 'get',
            responseType: 'blob',
            params: params
        }).then(function (rs) {
            let rsData = rs.data;
            if (callback && typeof callback == 'function') {
                callback(rsData);
            }
        }).catch(function (error) {
            if (error.response) {
                ElMessageBox.alert("服务器或网络错误！");
            } else if (error.request) {
                ElMessageBox.alert("请求错误！");
                console.log(error.request);
            } else {
                console.log('Error', error.message);
            }
        });
    },
    
    postFile: function (url, formData, success, failure) {
        axios.post(url, formData).then(res => {
            success(res);
        }).catch(e => {
            console.error(e);
            failure(e.message);
        });
    }
}

export default httpUtil;
