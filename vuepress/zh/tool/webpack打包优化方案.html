SplitChunksPlugin抽取公共代码

查看
// webpack.config.js
module.exports = {
  // ...
  optimization: {  
    splitChunks: {  
      chunks: 'all',  
      cacheGroups: {  
        vendor: {  
          test: /[\\/]node_modules[\\/]/,  
          name: 'vendors',  
          chunks: 'all'  
        },  
        common: {  
          name: 'common',  
          minChunks: 2,  
          priority: -20,  
          reuseExistingChunk: true  
        }  
      }  
    }  
};

Webpack允许使用ES6的import()函数来动态加载模块。

<!DOCTYPE html>  
<html lang="en">  
<head>  
  <meta charset="UTF-8">  
  <meta name="viewport" content="width=device-width, initial-scale=1.0">  
  <title>Webpack Code Splitting</title>  
</head>  
<body>  
  <button id="loadFeature">Load Feature</button>  
  <script>  
    const button = document.querySelector('#loadFeature');  
    button.addEventListener('click', () => {  
      import('./feature.js').then((module) => {  
        module.default(); // 调用模块中的默认导出函数  
      }).catch((err) => {  
        console.error('Failed to load feature:', err);  
      });  
    });  
  </script>  
</body>  
</html>

多入口文件分割
通过配置Webpack的entry属性，可以指定多个入口文件，从而实现代码分割。每个入口文件都会生成一个对应的输出文件。

module.exports = {  
  entry: {  
    main: './src/index.js',  
    login: './src/login.js'  
  },  
  output: {  
    filename: '[name].js',  
    path: path.resolve(__dirname, 'dist')  
  }  
};

路由按需加载
确保只有在用户访问特定路由时才加载相应的代码块。

const routes = [  
  {  
    path: '/',  
    component: () => import('./components/Home.vue')  
  },  
  {  
    path: '/about',  
    component: () => import('./components/About.vue')  
  }  
];  
  
const router = new VueRouter({  
  routes

# 5. 配置缓存

Webpack 的缓存配置主要通过 output 配置项中的 filename 和 chunkFilename 属性，结合内容哈希值（如 [contenthash]）来实现

查看
const path = require('path');  
  
module.exports = {  
  // 入口文件  
  entry: './src/index.js',  
    
  // 输出配置  
  output: {  
    // 输出目录  
    path: path.resolve(__dirname, 'dist'),  
      
    // 输出文件名，使用 [contenthash] 来确保文件内容变化时文件名也会变化  
    filename: '[name].[contenthash].js',  
      
    // 非入口 chunk 的文件名（如按需加载的模块）  
    chunkFilename: '[name].[contenthash].js',  
      
    // 清理 /dist 文件夹下的旧文件在每次构建之前  
    clean: true,  
  },  
    
  // 其他配置...  
    
  // 优化配置  
  optimization: {  
    // 拆分代码，提高缓存利用率  
    splitChunks: {  
      chunks: 'all', // 拆分所有类型的代码块  
      cacheGroups: {  
        vendor: {  
          test: /[\\/]node_modules[\\/]/, // 匹配 node_modules 目录下的模块  
          name: 'vendors', // 提取出的文件名  
          chunks: 'all', // 作用于异步和非异步代码块  
          priority: -10, // 优先级，数字越大优先级越高  
        },  
        // 可以添加更多的缓存组...  
      },  
    },  
      
    // 运行时代码分割  
    runtimeChunk: 'single', // 将运行时代码拆分为一个单独的 chunk  
  },  
    
  // 插件配置...  
    
  // 开发服务器配置（如果适用）  
  // devServer: {  
  //   // 配置开发服务器选项...  
  // },  
};
