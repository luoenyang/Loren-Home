// 引入xlsx.core.min.js
// 相关资源文件在public/src
<script src="/public/libs/xlsx.core.min.js" type="text/javascript"></script>


// 监听上传文件变化
function fileChange(){
    this.parseExcel();
}

function parseExcel(file) {
    // 分析头信息是否是excel
    let _this = this;
    if (
        file.raw.type != 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        && file.raw.type != 'application/vnd.ms-excel'
    ) {
        this.$notify.error({
            title: '错误',
            message: '请上传Excel类型的文件'
        });
        this.$refs.uploadFile.clearFiles();
        this.importloading = false;
        return false
    }


    console.log('开始加载Excel数据')
    let reader = new FileReader()
    let rABS = typeof FileReader !== "undefined" && (FileReader.prototype || {}).readAsBinaryString
    // 重点是下面的 file.raw
    if (rABS) {
        reader.readAsBinaryString(file.raw)
    } else {
        reader.readAsArrayBuffer(file.raw)
    }
    reader.onload = function (e) {
        let data = e.target.result;
        if (!rABS) {
            data = new Uint8Array(data)
        }
        let workBook = XLSX.read(data, { type: rABS ? 'binary' : 'array' });
        
        // 判断ecsel工作表个数
        if (workBook.SheetNames.length > 1) {
            _this.importloading = false;
            _this.$refs['uploadFile'].clearFiles();
            _this.$alert('excel格式错误，只允许有一个工作表/页签！', '提示');
            return;
        }

        // 遍历每个工作表数据
        workBook.SheetNames.forEach(name => {
            let sheet = workBook.Sheets[name]
            let json = XLSX.utils.sheet_to_json(sheet, {
                raw: false,
                header: 1
            })

            // 打印exceljson数据
            console.log(json)
        })
    }
},

    let tableData = JSON.parse(JSON.stringify(this.tableData));
    // 可进行数据处理在导出
    // ... 

    if (tableData.length > 0) {
        this.$confirm('是否把转换后的数据导出至excel?', {
            type: 'info'
        }).then(() => {

            let excelData = tableData;

            // 忽略头部序号
            let ws = XLSX.utils.json_to_sheet(excelData,{ skipHeader: true }); 

            // 设置excel列宽
            var wscols = [
                { wch: 10 },
                { wch: 60 },
                { wch: 30 },
                { wch: 20 },
                { wch: 30 },
                { wch: 30 },
                { wch: 30 },
                { wch: 30 },
                { wch: 30 },
            ];

            ws['!cols'] = wscols;

            // 合并单元格
            var merges = [{
                s: {//s为开始
                    c: 0,//开始列
                    r: 0//可以看成开始行,实际是取值范围
                },
                e: {//e结束
                    c: 0,//结束列
                    r: 1//结束行
                }
            }]
            ws["!merges"] = merges;

            let wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "Sheet1");
            XLSX.writeFile(wb, "数据转换导出.xlsx", { bookType: "xlsx", bookSST: true, type: 'base64' });
        })
    }
    else {
        this.$alert('请先导入数据进行转换', {
            type: 'warning'
        })
    }
},
